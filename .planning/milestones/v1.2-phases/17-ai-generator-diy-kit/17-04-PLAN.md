export async function POST(req: Request) {
  const session = await auth();
  if (!session?.user) return Response.json({ error: "Unauthorized" }, { status: 401 });
  if (!canManageTemplates(session.user.role)) return Response.json({ error: "Forbidden" }, { status: 403 });

  const body = await req.json();
  const parsed = someZodSchema.safeParse(body);
  if (!parsed.success) return Response.json({ error: "...", details: ... }, { status: 400 });

  return await withTenantContext(session.user.tenantId, async () => { ... });
}
